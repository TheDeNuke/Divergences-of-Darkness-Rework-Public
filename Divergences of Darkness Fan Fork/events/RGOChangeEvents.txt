province_event = {
	id = 199000
	title = "EVTNAME199000" #$PROVINCENAME$ Mine Runs Dry
	desc = "EVTDESC199000"

	trigger = {
		owner = { not = { has_country_flag = no_rgo_change_events } }
		owned_by = THIS
		controlled_by = THIS
		OR = { 
			trade_goods = coal
			trade_goods = iron
			trade_goods = copper
		}
		owner = {
			war = no
			civilized = yes
		}
		NOT = {
			terrain = hills
			terrain = dry_forest_hills
			terrain = deciduous_forest_hills
			terrain = conifer_forest_hills
			terrain = boreal_forest_hills
			terrain = jungle_hills
			terrain = seasonal_jungle_hills
			terrain = semi_desert_hills
			terrain = desert_hills

			terrain = tundra_hills

			terrain = deciduous_forest_mountains
			terrain = conifer_forest_mountains
			terrain = boreal_forest_mountains
			terrain = jungle_mountains
			terrain = seasonal_jungle_mountains
			terrain = semi_desert_mountains
			terrain = desert_mountains
			terrain = cold_desert_mountains
			terrain = tundra_mountains
			terrain = tundra
			terrain = cold_desert
		}
		NOT = {
			terrain = steppe
			terrain = dry_grassland
			terrain = mountain_grassland
			terrain = hilly_farmland
			terrain = mountain_farmland
		}
	}

	mean_time_to_happen = {
		months = 8000
	}

	option = {
		name = "Blast!"
        random_list = {
		   40 = { trade_goods = grain }
		   30 = { trade_goods = fruit }
           30 = { trade_goods = cotton }      
        }
	}
}

province_event = {
	id = 199005
	title = "EVTNAME199005" #Prospector Strikes Coal!
	desc = "EVTDESC199005"
	
	trigger = {
		owner = { not = { has_country_flag = no_rgo_change_events } }
		owned_by = THIS
		controlled_by = THIS
		province_control_days = 1095
        owner = {
            war = no
            year = 1850
            OR = {
                any_owned_province = {
                    trade_goods = coal
                    NOT = { unemployment_by_type = { value = 0.01 type = labourers } }
                }
                any_neighbor_country = {
                    war = no
                    any_owned_province = {
                        trade_goods = coal
                        NOT = { unemployment_by_type = { value = 0.01 type = labourers } }
                    }
                }
                any_greater_power = {
                    war = no
                    any_owned_province = {
                        trade_goods = coal
                        NOT = { unemployment_by_type = { value = 0.01 type = labourers } }
                    }
                }
            }
        }
		NOT = {
			trade_goods = precious_metal
			trade_goods = precious_goods 
			trade_goods = coal
			has_province_modifier = intense_farming
			has_province_modifier = exhausted_land
			has_province_modifier = switched_production
			has_province_modifier = new_production
		}
		OR = {
			AND = {
				unemployment_by_type = { type = farmers value = 0.5  }
				has_pop_type = farmers
				work_available = { worker = farmers }
			}
			AND = {
				unemployment_by_type = { type = labourers value = 0.5  }
				has_pop_type = labourers
				work_available = { worker = labourers }
			}
		}
		OR = {
			terrain = hills
			terrain = hilly_farmland
			terrain = dry_forest_hills
			terrain = deciduous_forest_hills
			terrain = conifer_forest_hills
			terrain = boreal_forest_hills
			terrain = jungle_hills
			terrain = seasonal_jungle_hills
			terrain = semi_desert_hills
			terrain = desert_hills
			terrain = cold_desert_hills
			terrain = tundra_hills
			terrain = mountain
			terrain = mountain_farmland
			terrain = dry_forest_mountains
			terrain = deciduous_forest_mountains
			terrain = conifer_forest_mountains
			terrain = boreal_forest_mountains
			terrain = jungle_mountains
			terrain = seasonal_jungle_mountains
			terrain = semi_desert_mountains
			terrain = desert_mountains
			terrain = cold_desert_mountains
			terrain = tundra_mountains
		}
	}

	mean_time_to_happen = {
		months = 2000
        modifier = {
            factor = 1.5
            OR = {
			   terrain = dry_forest_mountains
			   terrain = deciduous_forest_mountains
			   terrain = conifer_forest_mountains
			   terrain = boreal_forest_mountains
			   terrain = jungle_mountains
			   terrain = seasonal_jungle_mountains
			   terrain = tundra_mountains
            }
        }
		
        modifier = {
            factor = 0.001
            NOT = { trade_goods = cotton }
            NOT = { trade_goods = dye }
            NOT = { trade_goods = wool }
            NOT = { trade_goods = silk }
            NOT = { trade_goods = coal }
            NOT = { trade_goods = sulphur }
            NOT = { trade_goods = iron }
            NOT = { trade_goods = timber }
            NOT = { trade_goods = tropical_wood }
            NOT = { trade_goods = rubber }
            NOT = { trade_goods = oil }
            NOT = { trade_goods = precious_metal }
            NOT = { trade_goods = precious_goods }
            NOT = { trade_goods = cattle }
            NOT = { trade_goods = fish }
            NOT = { trade_goods = fruit }
            NOT = { trade_goods = grain }
            NOT = { trade_goods = tobacco }
            NOT = { trade_goods = tea }
            NOT = { trade_goods = coffee }
            NOT = { trade_goods = opium }
			NOT = { trade_goods = sugar }
			NOT = { trade_goods = spices }
			NOT = { trade_goods = horses }
			NOT = { trade_goods = copper }		
			NOT = { trade_goods = lead }
        }
	}

	option = {
		name = "They should get to work immediately."
		trade_goods = coal
		add_province_modifier = {
			name = switched_production
			duration = 1825
		}
		ai_chance = { factor = 100 }
	}
	
	option = {
		name = "No-- $PROVINCENAME$'s current production is too valuable."
		ai_chance = { factor = 0 }
	}
}
